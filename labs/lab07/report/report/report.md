using BenchmarkTools

S = 100.;
T = 1;

function createPath(S::Float64, r::Float64, sigma::Float64, T::Int64, n::Int64)
    h = T / n
    u = exp(r*h + sigma*sqrt(h))
    d = exp(r*h - sigma*sqrt(h))
    # –í–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å —Ç–æ–≥–æ, —á—Ç–æ —Ü–µ–Ω–∞ –∞–∫—Ü–∏–∏ –ø–æ–¥–Ω–∏–º–µ—Ç—Å—è

    Price = [S]
    s = S

        q = rand()
        if q < p
            s = S*u
            push!(Price, s)
        else
            s = S*d
            push!(Price, s)
        end
    end
    return Price
end
